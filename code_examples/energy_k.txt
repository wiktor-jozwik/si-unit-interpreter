// Energia kinetyczna - Ek = mv^2 / 2 

unit J: [kg*m^2/s^2]

let mass: [kg] = 10

let duration: [s] = 10
let distance: [m] = 20

let constModifier: [] = 2

let speed: [m/s] = distance / duration
let energy: [J] = mass * speed^2 / constModifier

let printoutString: string = "Energy equals: "
print(printoutString)
print(energy) // Should print Energy equals: ... J
